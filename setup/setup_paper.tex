	\newcommand{\addac}[2]{
		\DeclareAcronym{#1}{
			short = \ensuremath{#1},
			long  = {#2},
			sort  = {#1},
			class = nomencl
		}
	}
	\newcommand{\addab}[2]{
		\DeclareAcronym{#1}{
			short = {#1} ,
			long  = {#2},
			sort  = {#1},
			class = abbrev
		}
	}
